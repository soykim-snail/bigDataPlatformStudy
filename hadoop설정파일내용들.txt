export JAVA_HOME=/usr/local/java
export HADOOP_HOME=/root/hadoop-2.7.7
export HADOOP_HEAPSIZE=500

8. mapred_env.sh 파일 끝에 다음 내용을 추가한다.
export JAVA_HOME=/usr/local/java
export HADOOP_JOB_HISTORYSERVER_HEAPSIZE=500
export HADOOP_HOME=/root/hadoop-2.7.7

export HADOOP_HOME=/root/hadoop-2.7.7
export YARN_HEAPSIZE=500
